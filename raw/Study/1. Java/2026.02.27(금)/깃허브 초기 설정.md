\# cd(change directory) : 디렉토리 이동 명령어
cd D:/java_project/MyJava

\# 이 디렉토리는 안전한 디렉토리입니다.

\# 깃허브에 자료 올리는 사람의 이름
git config \-\-global user.name "bluesky"

\# 깃허브에 자료 올리는 사람의 이메일 주소
git config \-\-global user.email bluesky@naver.com
