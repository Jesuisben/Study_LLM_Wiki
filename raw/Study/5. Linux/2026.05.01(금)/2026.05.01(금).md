- dockerfile과 docker compose의 차이 (P.91)
특징 							dockerfile 												docker compose
목적 							단일 컨테이너 이미지 정의 					여러 컨테이너를 조합하여 애플리케이션 구성
파일 형식 					Dockerfile(텍스트 형식) 							YAML(docker-compose.yml)
사용 대상 					컨테이너 이미지 생성 							서비스 간 관계 설정 및 실행
주요 명령어 				FROM, RUN, CMD, COPY, EXPOSE 등 	services, build, ports, volumes, networks
기능 							애플리케이션의 실행 환경 정의 				컨테이너 오케스트레이션
단독 실행 가능 여부 	가능 (단일 컨테이너 실행에 적합) 			불가능 (단독으로 컨테이너 빌드 및 실행 불가)

2) 위의 정보들을 이용해서 메니페스트 파일 작성
이름 : compose_mysql_springboot.yml

내용 :

```yaml
services: # 서비스는 컨테이너라고 생각하면 된다고 하심
  mysql-svc: # mysql 컨테이너 / 하위에는 해당 컨테이너의 설정을 작성
    container_name: mysql-ctr
    image: mysql:8.0
    networks:
      - spring-mysql-net
    volumes:
      - mysql-spring-vol:/var/lib/mysql
    restart: always
    command: --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci
    environment:
      MYSQL_ROOT_PASSWORD: My@Sql01
      MYSQL_DATABASE: coffee
      MYSQL_USER: spring
      MYSQL_PASSWORD: Spring@1234
    ports:
      - "3306:3306"

  spring-svc: # spring 컨테이너
    container_name: spring-ctr
    depends_on: # 의존/참조 관계를 나타냄
      - mysql-svc
    image: myspring-img:latest # <-- 사용자 정의 이미지 이름 (Dockerfile로 만들어서 준비한 이미지)
    networks:
      - spring-mysql-net
    ports:
      - "9000:9000"
    restart: always
    environment:
      SPRING_PROFILES_ACTIVE: docker
      SPRING_DATASOURCE_URL: jdbc:mysql://mysql-svc:3306/coffee
      SPRING_DATASOURCE_USERNAME: spring
      SPRING_DATASOURCE_PASSWORD: Spring@1234

networks:
  spring-mysql-net:

volumes:
  mysql-spring-vol:


```

- 매니페스트 파일 작성
(교육 자료 - dk04.docker_file - compose02 - compose_mysql_wordpress.yml)

해당 파일이 위에 주어진 정보랑 다르고 들여쓰기가 잘못되어 정상적으로 컴포즈로 실행이 안됨(중요!!!!)
추가적으로 매니페스트 파일에 위에서 주어진 정보가 아닌 다른 정보들이 적혀있어서 수정함
그리고 수정한 파일만 다시 넣음
(교육 - 5. Linux - 2026.05.01(금) - 매니페스트 파일 - )
해당 폴더 안에 파일 들여쓰기만 수정한 매니페스트 파일과 내용 자체를 수정한 파일 둘다 있음

(아래의 매니페스트 파일 내용은 "compose_mysql_wordpress.yml 실습 전체 내용 수정본" 폴더에 있는 파일)

```yaml
services:
  mysql01: # 서비스 이름 (컴포즈의 컨테이너?) (depends_on과 같이 참조할때 사용하는 이름)
    container_name: mysql01  # 실질적인 컨테이너 이름
    image: mysql:8.0
    networks:
      - network01
    volumes: # 이미 설정된 볼륨 이름가져오고 그에 맞는 경로 설정
      - volume_mysql:/var/lib/mysql
    restart: always
    command: --character-set-server=utf8mb4 --collation-server=utf8mb4_unicode_ci # 정석적인 설정법이기 때문에 알아만 두기
    environment:
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: coffee
      MYSQL_USER: root
      MYSQL_PASSWORD: My@Sql01
    ports: # MySQL 포트는 3306
      - "3306:3306"

  wordpress01:
    container_name: wordpress01
    depends_on: # DB 참조하기
      - mysql01
    image: wordpress
    networks: # 네트워크는 참조하는 것과 같은 네트워크를 써야 함
      - network01
    volumes:
      - volume_wordpress:/var/www/html
    ports: # 내 마음대로? 설정:http(80)
      - 8085:80
    restart: always
    environment:
      WORDPRESS_DB_HOST: mysql01
      WORDPRESS_DB_NAME: coffee
      WORDPRESS_DB_USER: root
      WORDPRESS_DB_PASSWORD: My@Sql01
networks:
  network01:
volumes:
  volume_mysql:
  volume_wordpress:


```
